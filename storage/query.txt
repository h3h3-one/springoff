PRAGMA foreign_keys=on;

CREATE TABLE IF NOT EXISTS albums(
    id_album INTEGER PRIMARY KEY AUTOINCREMENT,
    title_album VARCHAR(30) NOT NULL,
    path_album VARCHAR(30) NOT NULL,
    path_cover VARCHAR(30) NOT NULL
);

CREATE TABLE IF NOT EXISTS images(
     id_image INTEGER PRIMARY KEY AUTOINCREMENT,
     path_image VARCHAR(30) NOT NULL,
     title_album INTEGER NOT NULL,
     CONSTRAINT fk_title
     FOREIGN KEY (title_album)
     REFERENCES albums (title_album)
     ON DELETE CASCADE
);
